--문제 6) CLERK는 7% , OFFICER는 5% , MANAGER는 3%로 연봉을 인상하기로 했다. STAFF_SAL테이블을 사용하여 각 직원별 인상 연봉을 예상해보자 ( CASE WEHN 사용해보기 )
--(현재연봉을 기준으로 직위별 예상연봉을 구해보자)
select * from staff_sal;
select job ,
            case when  job = 'CLERK'  then to_char(current_sal * 1.07,'999,999')
                     when  job = 'OFFICER' then  to_char(current_sal * 1.05,'999,999')
                     when  job = ' MANAGER' then  to_char(current_sal * 1.03,'999,999')
            end   as 예상연봉 
from staff_sal;
