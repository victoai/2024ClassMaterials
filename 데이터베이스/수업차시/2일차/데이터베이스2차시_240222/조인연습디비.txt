  ocode varchar2(20) not null primary key,
  odate date,
  id varchar2(20),

SELECT   name, address, tel, odate, pname, sale_cnt ,price ,sale_cnt* price

// sql  조인 표준 

SELECT  name, address, tel, odate, pname, sale_cnt, price, sale_cnt * price
FROM   tbl_test_order o
JOIN     tbl_test_customer c
ON       o.id = c.id
JOIN     tbl_test_goods g
ON       o.pcode = g.pcode ;
