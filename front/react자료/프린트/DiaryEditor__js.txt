import React, {useRef , useState} from "react";

const  DiaryEditor = ( {onCreate})=>{
 const authorInput   = useRef();
 const contentTextarea = useRef();
 
 
 const[state, setState] = useState({
    author:"",
    content:"",
    emotion:1
 });

 
 const handlerChangeState= (e)=>{
   setState({
        //author: state.author,
        ...state,
        [e.target.name ]: e.target.value
 })}


 const handleSubmit = ()=>{

    if(state.author.length <1){
       // alert("한글자 이상");
        authorInput.current.focus();
        return;

    }

    if( state.content.length <5 ){
       // alert(" 글은 5글자 이상");
       contentTextarea.current.focus();
        return;
    }

    onCreate(state.author, state.content, state.emotion);
     alert("일기저장");
     console.log(  state)

     //초기화
     setState({
        author:"",
        content:"",
        emotion:""
     });
 }
 

    return(
        <div className="DiaryEditor"> 
            <h2>오늘의 일기장</h2>
            <div>
                <input ref={authorInput} name="author"  value={state.author}  onChange={handlerChangeState} />
            </div>

            <div>
            <textarea  ref={contentTextarea} name="content" value={state.content}  onChange={handlerChangeState}/>
            </div>

            <div>
                오늘의 감점점수
                <select    name="emotion" value={state.emotion} onChange={handlerChangeState} >
                    <option value={1}> 1</option>
                    <option value={2}> 2</option>
                    <option value={3}> 3</option>
                    <option value={4}> 4</option>
                    <option value={5}> 5</option>
                    
                </select>
            </div>
            <div><button  onClick={handleSubmit}>저장하기</button></div>
        </div>
    );
}

export default DiaryEditor;
