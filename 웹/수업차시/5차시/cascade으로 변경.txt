CREATE TABLE 자식테이블 (
    자식ID NUMBER PRIMARY KEY,
    부모ID NUMBER,
    FOREIGN KEY (부모ID) REFERENCES 부모테이블(부모ID) ON DELETE CASCADE
);

-- 새로운 외래 키 제약 조건 추가 (CASCADE 옵션 포함)
